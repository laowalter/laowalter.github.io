Start-stop-daemon /etc/init.d/jupyter

#!/sbin/openrc-run

depend() {
  need net
  use dns logger
}

PIDFILE="/var/run/${SVCNAME}.pid"
PYENV="/home/walter/.virtualenv/python3/bin/activate"
NAME="walter"
CMD="/home/walter/.virtualenv/python3/bin/jupyter-notebook"
OPTS="--no-browser"
WORKING_DIR="/home/walter/chan"

start() {
    ebegin "Starting jupyter"
    source $PYENV
    start-stop-daemon --start --background \
    --user $NAME \
    --chdir $WORKING_DIR \
    --make-pidfile --pidfile $PIDFILE \
    --exec $CMD \
    -- $OPTS
    eend $?
}


stop() {
    ebegin "Stopping jupyter"
    start-stop-daemon --stop \
    --pidfile $PIDFILE \
    --user $NAME \
    --exec $CMD \
    eend $?
}
