使用技巧

  • 文件编码

    :set fenc=gbk
:set fenc=utf-8

  • DOS UNIX 转换

    set ff=unix

  • Swap character

    xp
Xp

  • Global Replace, $ says “put vim in last-line mode”.

    :1,$s/George Bush/Barack Obama/g 
:1,$s/td align="CENTER"/td/g

  • Delete blank lines in vim (empty lines), use this vim command in “last line mode”

    :g/^$/d

  • 折叠快捷键

    • zc 折叠，只折叠最外层的折叠
    • zC 对所在范围内所有嵌套的折叠点进行折叠，包括嵌套的所有折叠.
    • zo 展开折叠，只展开最外层的折叠.
    • zO 对所在范围内所有嵌套的折叠点展开，包括嵌套折叠.
    • [z 到当前打开的折叠的开始处。
    • ]z 到当前打开的折叠的末尾处。
    • zj 向下移动。到达下一个折叠的开始处。关闭的折叠也被计入。
    • zk 向上移动到前一折叠的结束处。关闭的折叠也被
    • zd 删除 (delete) 在光标下的折叠。仅当 ‘foldmethod’ 设为 “manual” 或 “marker” 时有效。
    • zD 循环删除 (Delete) 光标下的折叠，即嵌套删除折叠。
    • 仅当 ‘foldmethod’ 设为 “manual” 或 “marker” 时有效。
    • zE 除去 (Eliminate) 窗口里“所有”的折叠。
    • 仅当 ‘foldmethod’ 设为 “manual” 或 “marker” 时有效。
    • zfap 将光标移到段落内，然后按zfap，就可以自动对整个段落添加折叠标签
    • zr 假定你已经创建了若干折叠，而现在需要阅览全部文本。你可以移到每个折叠处，并键入”zo”。若要做得更快，可以用这个命令:zr
    • zm 这将折叠更多 (M-ore)。你可以重复 “zr” 和 “zm” 来打开和关闭若干层嵌套的折叠，不然得一个一个的用zc来折叠. 如果你有一个嵌套了好几层深的折叠，你可以用这个命令把它们全部打开:
    • zR 这将减少折叠直至一个也不剩。而用下面这个命令你可以关闭所有的折叠:
    • zM 这将增加折叠，直至所有的折叠都关闭了。
    • 你可以用 |zn| 命令快速禁止折叠功能。然后 |zN| 恢复原来的折叠。|zi| 切换于两者之间。

Plugin Installation

  1. Install to ~/.vim/autoload/pathogen.vim. Or copy and paste:

    mkdir -p ~/.vim/autoload ~/.vim/bundle && \
curl -LSso ~/.vim/autoload/pathogen.vim https://tpo.pe/pathogen.vim

  2. Install Vundle

    git clone http://github.com/gmarik/vundle.git ~/.vim/bundle/vundle

  3. Runtime Path Manipulation

%filename: ~/.vimrc

set nocompatible
filetype off
set  rtp+=~/.vim/bundle/vundle/
call vundle#begin()

Bundle 'Lokaltog/powerline', {'rtp': 'powerline/bindings/vim/'}
Bundle 'gmarik/vundle'
Bundle 'tpope/vim-surround'
Bundle 'jiangmiao/auto-pairs'
" Bundle 'Lokaltog/vim-powerline'
Bundle 'bling/vim-airline'
Bundle 'perl-support.vim'
Bundle 'Lokaltog/vim-easymotion'
Bundle 'desert256.vim'
Bundle 'solarized'
Bundle 'PHP-correct-Indenting'
Bundle 'mattn/emmet-vim'
Bundle 'php.vim-html-enhanced'
Bundle 'OmniCppComplete'
Bundle 'Railscasts-Theme-GUIand256color'
Plugin 'Javascript-Indentation'
Plugin 'Enhanced-Javascript-syntax'
Plugin 'jmcantrell/vim-virtualenv'
Plugin 'klen/python-mode'
call vundle#end()

" Powerline setup
set guifont=DejaVu\ Sans\ Mono\ for\ Powerline\ 9
set laststatus=2
set t_Co=256

let g:Powerline_symbols='unicode'

" Perl_Perltidy in Perl-support
let g:Perl_Perltidy  = 'on'

"set file encoding
set fileencodings=utf-8,gb2312,gbk,gb18030 

" Python-mode

" Maximum line length
let g:pymode_options_max_line_length = 120

" Auto check on save
let g:pymode_lint_write = 1

" syntax highlighting
let g:pymode_syntax = 1
let g:pymode_syntax_all = 1
let g:pymode_syntax_indent_errors = g:pymode_syntax_all
let g:pymode_syntax_space_errors = g:pymode_syntax_all

" Don't autofold code
let g:pymode_folding = 0

set cursorline
set number
syntax enable
set background=dark
colorscheme solarized

set tabstop=4 shiftwidth=4 expandtab

colorscheme railscasts

filetype plugin indent on
syntax on

inoremap <C-e> <Esc>A
inoremap <C-a> <Esc>I

set GuiFont and Size

%filename: .gvimrc

set guifont=TerminusTTF:h22

  • .vimrc

Then BundleInstall

  • Install ncurses

    touch /etc/portage/package.use/ncurses
echo "sys-libs/ncurses tinfo" > /etc/portage/package.use/ncurses
emerge sys-libs/ncurses:5

  • compile

    cd .vim/bundle/YouCompleteMe/
./install --all
