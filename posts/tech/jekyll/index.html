Jekyll Install

$ sudo emerge -av dev-ruby/rubygems
$ gem install bundler jekyll
$ echo "export PATH=$PATH:~/.gem/ruby/2.3.0/bin" > ~/.bashrc "
$ source ~/.bashrc
$ jekyll new blog
$ cd blog
$  bundle install
~/blog/ $ bundle exec jekyll serve
