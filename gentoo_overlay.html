Add local overlay

  • Create directory
mkdir -p /usr/local/portage/metadata
echo "masters = gentoo" > /usr/local/portage/metadata/layout.conf
chown -R portage:portage /usr/local/portage/
  • Modify make.conf
PORTDIR_OVERLAY="/usr/local/portage/ ${PORTDIR_OVERLAY}"
  • Add [local] part to /etc/portage/repos.conf/gentoo.conf 
[local]
location = /usr/local/portage
auto-syc = no

Add a local ebuild

sudo cp -r /usr/portage/net-p2p/transmission /usr/local/portage
cd /usr/local/portage
ebuild transmission-2.92-r2.ebuild manifest
