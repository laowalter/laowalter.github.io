1.0 Install /Upgrade php-fpm

1.1 Config /etc/php/fpm-php7.2/fpm.d/www.conf

FILE:/etc/php/fpm-php7.2/fpm.d/www.conf
  listen = /run/php-fpm.socket
  listen.owner = nginx
  listen.group = nginx
  listen.mode = 0660

1.2 Adjust upload limited

FILE:/etc/php/fpm-php7.2/php.ini
upload_tmp_dir = /tmp
upload_max_filesize = 20M

1.3 /etc/ngnix/nginx.conf

FILE:/etc/ngnix/nginx.conf
client_max_body_size 20M;
client_body_buffer_size 128k;

2.0 Add php-fpm to default service

sudo  /etc/init.d/php-fpm start
rc-update add php-fpm default

3.0 Setup php-fpm in nginx

FILE:/etc/ngnix/nginx.conf
server {
    listen 0.0.0.0:7001;
    server_name pannel;
    access_log /var/log/pannel.log main;
    error_log /var/log/pannel.info error;

    root /var/www/localhost/htdocs/pannel;
    location / {
            index index.html;
            fastcgi_index index.html;
    }
    location ~ \.php$ {
            try_files $uri =404;
            include /etc/nginx/fastcgi.conf;
            fastcgi_pass unix:/run/php-fpm.socket;
    }
