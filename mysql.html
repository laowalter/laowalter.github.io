MySQL基本操作

增加用户，并赋予权限

  1. 创建本机用户stock@localhost，不能从其他机器访问,即使grant all privileges 也没用
create user 'stock'@'localhost' identified by 'my_pass';
grant all privileges on *.* to 'stock'@'localhost' with grant option;
flush privileges;
  1. 创建用户stock@%，可以在任何机器访问此数据库
create user 'stock'%' identified by 'my_pass';
grant all privileges on *.* to 'stock'@'%' with grant option;
flush privileges;

赋予个别用户特别权限

mysql> CREATE USER 'custom'@'localhost' IDENTIFIED BY 'password';
mysql> GRANT SELECT,INSERT,UPDATE,DELETE,CREATE,DROP
    ->     ON bankaccount.*
    ->     TO 'custom'@'localhost';
mysql> CREATE USER 'custom'@'host47.example.com' IDENTIFIED BY 'password';
mysql> GRANT SELECT,INSERT,UPDATE,DELETE,CREATE,DROP
    ->     ON expenses.*
    ->     TO 'custom'@'host47.example.com';
mysql> CREATE USER 'custom'@'%.example.com' IDENTIFIED BY 'password';
mysql> GRANT SELECT,INSERT,UPDATE,DELETE,CREATE,DROP
    ->     ON customer.*
    ->     TO 'custom'@'%.example.com';
