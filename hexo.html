Run server

Autorun at boot in Mac

sudo vi /Library/LaunchDaemons/com.localhost.hexo.plist

<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE plist PUBLIC "-//Apple//DTD PLIST 1.0//EN" "http://www.apple.com/DTDs/PropertyList-1.0.dtd">
<plist version="1.0">
<dict>
  <key>Label</key>
  <string>com.localhost.hexo</string>
  <key>RunAtLoad</key>
  <true/>
    <key>ProgramArguments</key>
    <array>
    <string>/Volumes/Home/Walter/.nvm/versions/node/v5.7.0/bin/hexo</string>
    <string>-i</string>
    <string>192.168.0.99</string>
    <string>-p</string>
    <string>9000</string>
    </array>
</dict>
</plist>
