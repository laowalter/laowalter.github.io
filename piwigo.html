1.0 Use Piwigo 注意事项

所有的图片文件都放在 piwigo/upload/ 的目录中。

1.1 Gentoo Nginx / php-fm

1.1.1 config Nginx to support

FILE: /etc/nginx/nginx.conf
include /etc/nginx/conf.d/*.conf;
FILE: /etc/nginx/conf.d/piwigo.conf
server {
    listen 0.0.0.0:3128;
    server_name photo;
    #root /var/www/localhost/;
    root /var/www/localhost/htdocs/piwigo;


    location ~ \.php$ {
            try_files $uri =404;
            include /etc/nginx/fastcgi.conf;
            fastcgi_pass unix:/run/php-fpm.socket;
    }
}

1.1.2 Config php-fpm

1.1.2.1 config www.conf
FILE: /etc/php/fpm-php7.1/fpm.d/www.conf
  listen = /run/php-fpm.socket
  listen.owner = nginx
  listen.group = nginx
  listen.mode = 0660
1.1.2.2 adjust upload limited
FILE: /etc/php/fpm-php7.1/php.ini
upload_tmp_dir = /tmp
upload_max_filesize = 20M
FILE: /etc/ngnix/nginx.conf
client_max_body_size 20M;
client_body_buffer_size 128k;
1.1.2.3 add php-fpm to default service
sudo  /etc/init.d/php-fpm start
rc-update add php-fpm default

1.2 Freebsd /Freenas Nginx php-fpm

1.2.1 Install

pkg install nginx

1.2.2 php-fpm

FILE: /usr/local/etc/php-fpm.d/www.conf
user = www
group = www
listen = /var/run/php72-fpm.sock
listen.owner = www
listen.group = www
listen.mode = 0660

1.2.3 ngfix

FILE: /usr/local/etc/nginx/nginx.conf
worker_processes  1;

events {
    worker_connections  1024;
}


http {
    include       mime.types;
    default_type  application/octet-stream;

    sendfile        on;

    keepalive_timeout  65;

    server {
        listen 0.0.0.0;
        server_name piwigo;
        access_log  /var/log/nginx/access.log;
        error_log  /var/log/nginx/error.log;

        root /usr/local/www/piwigo;
        location / {
            index index.php;
        }
        location ~ [^/]\.php(/|$) {
            fastcgi_split_path_info ^(.+?\.php)(/.*)$;
            if (!-f $document_root$fastcgi_script_name) {
                    return 404;
            }

            fastcgi_param HTTP_PROXY "";

            fastcgi_pass unix:/var/run/php72-fpm.sock;
            fastcgi_index index.php;

            include fastcgi_params;

            fastcgi_param  SCRIPT_FILENAME   $document_root$fastcgi_script_name;
        }
    }
}
